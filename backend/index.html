import express from "express";
import { l402 } from "l402-kit";

const app = express();

app.get("/premium", l402({
  priceSats: 100,                          // ~$0.10 per call
  ownerLightningAddress: "you@blink.sv",  // receives 99.7%
}), (req, res) => {
  res.json({ data: "Payment confirmed ⚡" });
});

app.listen(3000);
